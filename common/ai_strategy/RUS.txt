keep_faction_with_fra = {
	allowed = { original_tag = FRA }

	enable = {
		country_exists = GER
		NOT = {
			has_completed_focus = RUS_continued_dialogue_with_the_germans
		}
		OR = {
			has_government = democratic
			has_government = neutrality
		}
		FRA = {
			OR = {
				has_government = democratic 
				has_government = neutrality
			}
			NOT = {
				has_completed_focus = FRA_attempt_german_reapproachment
				has_completed_focus = RUS_on_our_own
			}
		}
	}
	abort = {
		OR = {
			FRA = {
				has_completed_focus = FRA_attempt_german_reapproachment
			}
			ROOT = {
				OR = {
					has_completed_focus = RUS_continued_dialogue_with_the_germans
					has_completed_focus = RUS_on_our_own
				}
			}
		}
	}

	ai_strategy = {
		type = alliance
		id = "FRA"
		value = 200
	}
}

protect_serbia_against_austria = {
	allowed = { original_tag = RUS }

	enable = {
		has_global_flag = Franz_ferdinand_shot
		AUS = { NOT = { has_war_with = SER } }
		country_exists = SER
	}
	abort = {
		OR = {
			NOT = { country_exists = SER }
			has_war_with = SER
		}
	}

	ai_strategy = {
		type = protect
		id = "SER"
		value = 200
	}
	ai_strategy = {
		type = alliance
		id = "SER"
		value = 200
	}
	ai_strategy = {
		type = diplo_action_desire
		id = SER
		target = guarantee
		value = 20
	}
}
