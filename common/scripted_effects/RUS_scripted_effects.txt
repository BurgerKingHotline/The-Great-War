RUS_civil_war_start = {
	country_event = { id = rus_civilwar.4 }
	news_event = { id = TLG_newsevents.2 hours = 4}
	SOV = {
		set_cosmetic_tag = RUS
		
		set_politics = {
			ruling_party = communism
			elections_allowed = no
			long_name = RUS_communism_party_long
			name = RUS_communism_party
		}
	}
	set_global_flag = Russian_civil_war_flag
}

}

RUS_increase_intervention_campaign_cost_effect = {

}

RUS_industrialisation_up = {
	if = {
		limit = {
			has_idea = RUS_industrialising_nation_1
		}
		swap_ideas = {
			remove_idea =RUS_industrialising_nation_1
			add_idea = RUS_industrialising_nation_2
		}
	}
	else_if = {
		limit = {
			has_idea = RUS_industrialising_nation_2
		}
		swap_ideas = {
			remove_idea = RUS_industrialising_nation_2
			add_idea = RUS_industrialising_nation_3
		}
	}
	else_if = {
		limit = {
			has_idea = RUS_industrialising_nation_3
		}
		swap_ideas = {
			remove_idea = RUS_industrialising_nation_3
			add_idea = RUS_industrialising_nation_4
		}
	}
    else_if = {
		limit = {
			has_idea = RUS_industrialising_nation_4
		}
		swap_ideas = {
			remove_idea = RUS_industrialising_nation_4
			add_idea = RUS_industrialising_nation_5
		}
	}
	else_if = {
		limit = {
			has_idea = RUS_industrialising_nation_5
		}
		swap_ideas = {
			remove_idea = RUS_industrialising_nation_5
			add_idea = RUS_industrialising_nation_6
		}
	}
	else_if = {
		limit = {
			has_idea = RUS_industrialising_nation_6
		}
		random_owned_controlled_state = {
			add_building_construction = {
				type = industrial_complex
				level = 2
				instant_build = yes
			}
			add_extra_state_shared_building_slots = 2
		}
		random_owned_controlled_state = {
			add_building_construction = {
				type = industrial_complex
				level = 2
				instant_build = yes
			}
			add_extra_state_shared_building_slots = 2
		}
		add_stability = -0.05
	}
	add_to_variable = {
		var = RUS_socialist_radicalisation
		value = 10
		tooltip = RUS_socialist_radicalisation_increase
	}
	add_popularity = {
		ideology = communism
		popularity = 0.05
	}
}

RUS_industrialisation_down = {
	if = {
		limit = {
			has_idea = RUS_industrialising_nation_6
		}
		swap_ideas = {
			remove_idea = RUS_industrialising_nation_6
			add_idea = RUS_industrialising_nation_5
		}
	}
	else_if = {
		limit = {
			has_idea = RUS_industrialising_nation_5
		}
		swap_ideas = {
			remove_idea = RUS_industrialising_nation_5
			add_idea = RUS_industrialising_nation_4
		}
	}
	else_if = {
		limit = {
			has_idea = RUS_industrialising_nation_4
		}
		swap_ideas = {
			remove_idea = RUS_industrialising_nation_4
			add_idea = RUS_industrialising_nation_3
		}
	}
	else_if = {
		limit = {
			has_idea = RUS_industrialising_nation_3
		}
		swap_ideas = {
			remove_idea = RUS_industrialising_nation_3
			add_idea = RUS_industrialising_nation_2
		}
	}
	else_if = {
		limit = {
			has_idea = RUS_industrialising_nation_2
		}
		swap_ideas = {
			remove_idea = RUS_industrialising_nation_2
			add_idea = RUS_industrialising_nation_1
		}
	}
	else_if = {
		limit = {
			has_idea = RUS_industrialising_nation_1
		}
        add_stability = 0.05
		add_political_power = 50
	}
    else_if = { 
        limit = {
            NOT = {
                OR = {
                    has_idea = RUS_industrialising_nation_1
                    has_idea = RUS_industrialising_nation_2
                    has_idea = RUS_industrialising_nation_3
                    has_idea = RUS_industrialising_nation_4
                    has_idea = RUS_industrialising_nation_5
                }
            }
        }
        add_stability = 0.05
		add_political_power = 50
    }
	add_to_variable = {
		var = RUS_socialist_radicalisation
		value = -10
		tooltip = RUS_socialist_radicalisation_increase
	}
	add_popularity = {
		ideology = communism
		popularity = -0.05
	}
}
