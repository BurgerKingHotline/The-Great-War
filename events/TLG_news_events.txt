news_event = { #franco-german peace deal - Germans win
	id = TLG_newsevents.4
	title = TLG_newsevents.4.t	
	desc = {
		trigger = {
			OR = {
				RUS = {
					has_war_with = GER
				}
				ENG = {
					has_war_with = GER
				}
			}
		}
		text = TLG_newsevents.4.d1
	}
	desc = {
		trigger = {
			NOT = {
				OR = {
					RUS = {
						has_war_with = GER
					}
					ENG = {
						has_war_with = GER
					}
				}
			}
		}
		text = TLG_newsevents.4.d2
	}

	is_triggered_only = yes
	major = yes
	
	option = { #lessgooo
		name = TLG_newsevents.4.a
		trigger = {
			is_in_faction_with = GER
		}
	}
	option = { #german hegemony is secured
		name = TLG_newsevents.4.b
		trigger = {
			NOT = { is_in_faction_with = GER }
			NOT = { tag = FRA }
		}
	}
	option = { #we have been humiliated again
		name = TLG_newsevents.4.b
		trigger = {
			tag = FRA
		}
		hidden_effect = {
			add_popularity = {
				ideology = fascism
				popularity = 0.03
			}
		}
	}
}
